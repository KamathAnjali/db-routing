This is a Database called e_commerce.
This Database consists of 8 tables.

These tables are:

1. Products (displayed as: products)

2. Customers (displayed as: customers)

3. Customer_Payment_Methods (displayed as: customer payment methods)

4. Invoices (displayed as: invoices)

5. Orders (displayed as: orders)

6. Order_Items (displayed as: order items)

7. Shipments (displayed as: shipments)

8. Shipment_Items (displayed as: shipment items)


Table 1: Tablename is Products
Display name: products
DDL query for Products:
CREATE TABLE `Products` (
`product_id` INTEGER PRIMARY KEY ,
`parent_product_id` INTEGER,
`product_name` VARCHAR(80),
`product_price` DECIMAL(19,4) DEFAULT 0,
`product_color` VARCHAR(50),
`product_size` VARCHAR(50),
`product_description` VARCHAR(255)
)


Table 2: Tablename is Customers
Display name: customers
DDL query for Customers:
CREATE TABLE `Customers` (
`customer_id` INTEGER PRIMARY KEY,
`gender_code` VARCHAR(1) NOT NULL,
`customer_first_name` VARCHAR(50),
`customer_middle_initial` VARCHAR(1),
`customer_last_name` VARCHAR(50),
`email_address` VARCHAR(255),
`login_name` VARCHAR(80),
`login_password` VARCHAR(20),
`phone_number` VARCHAR(255),
`address_line_1` VARCHAR(255),
`town_city` VARCHAR(50),
`county` VARCHAR(50),
`country` VARCHAR(50)
)


Table 3: Tablename is Customer_Payment_Methods
Display name: customer payment methods
DDL query for Customer_Payment_Methods:
CREATE TABLE `Customer_Payment_Methods` (
`customer_id` INTEGER NOT NULL,
`payment_method_code` VARCHAR(10) NOT NULL,
FOREIGN KEY (`customer_id` ) REFERENCES `Customers`(`customer_id` )
)


Table 4: Tablename is Invoices
Display name: invoices
DDL query for Invoices:
CREATE TABLE `Invoices` (
`invoice_number` INTEGER PRIMARY KEY,
`invoice_status_code` VARCHAR(10) NOT NULL,
`invoice_date` DATETIME
)


Table 5: Tablename is Orders
Display name: orders
DDL query for Orders:
CREATE TABLE `Orders` (
`order_id` INTEGER PRIMARY KEY,
`customer_id` INTEGER NOT NULL,
`order_status_code` VARCHAR(10) NOT NULL,
`date_order_placed` DATETIME NOT NULL,
FOREIGN KEY (`customer_id` ) REFERENCES `Customers`(`customer_id` )
)


Table 6: Tablename is Order_Items
Display name: order items
DDL query for Order_Items:
CREATE TABLE `Order_Items` (
`order_item_id` INTEGER PRIMARY KEY ,
`product_id` INTEGER NOT NULL,
`order_id` INTEGER NOT NULL,
`order_item_status_code` VARCHAR(10) NOT NULL,
FOREIGN KEY (`product_id` ) REFERENCES `Products`(`product_id` ),
FOREIGN KEY (`order_id` ) REFERENCES `Orders`(`order_id` )
)


Table 7: Tablename is Shipments
Display name: shipments
DDL query for Shipments:
CREATE TABLE `Shipments` (
`shipment_id` INTEGER PRIMARY KEY,
`order_id` INTEGER NOT NULL,
`invoice_number` INTEGER NOT NULL,
`shipment_tracking_number` VARCHAR(80),
`shipment_date` DATETIME,
FOREIGN KEY (`invoice_number` ) REFERENCES `Invoices`(`invoice_number` ),
FOREIGN KEY (`order_id` ) REFERENCES `Orders`(`order_id` )
)


Table 8: Tablename is Shipment_Items
Display name: shipment items
DDL query for Shipment_Items:
CREATE TABLE `Shipment_Items` (
`shipment_id` INTEGER NOT NULL,
`order_item_id` INTEGER NOT NULL,
PRIMARY KEY (`shipment_id`,`order_item_id`),
FOREIGN KEY (`shipment_id` ) REFERENCES `Shipments`(`shipment_id` ),
FOREIGN KEY (`order_item_id` ) REFERENCES `Order_Items`(`order_item_id` )
)

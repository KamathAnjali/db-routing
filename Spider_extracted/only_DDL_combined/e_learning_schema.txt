This is a Database called e_learning.
This Database consists of 6 tables.

These tables are:

1. Course_Authors_and_Tutors (displayed as: course authors and tutors)

2. Students (displayed as: students)

3. Subjects (displayed as: subjects)

4. Courses (displayed as: courses)

5. Student_Course_Enrolment (displayed as: student course enrolment)

6. Student_Tests_Taken (displayed as: student tests taken)


Table 1: Tablename is Course_Authors_and_Tutors
Display name: course authors and tutors
DDL query for Course_Authors_and_Tutors:
CREATE TABLE `Course_Authors_and_Tutors` (
`author_id` INTEGER PRIMARY KEY,
`author_tutor_ATB` VARCHAR(3),
`login_name` VARCHAR(40),
`password` VARCHAR(40),
`personal_name` VARCHAR(80),
`middle_name` VARCHAR(80),
`family_name` VARCHAR(80),
`gender_mf` VARCHAR(1),
`address_line_1` VARCHAR(80)
)


Table 2: Tablename is Students
Display name: students
DDL query for Students:
CREATE TABLE `Students` (
`student_id` INTEGER PRIMARY KEY,
`date_of_registration` DATETIME,
`date_of_latest_logon` DATETIME,
`login_name` VARCHAR(40),
`password` VARCHAR(10),
`personal_name` VARCHAR(40),
`middle_name` VARCHAR(40),
`family_name` VARCHAR(40)
)


Table 3: Tablename is Subjects
Display name: subjects
DDL query for Subjects:
CREATE TABLE `Subjects` (
`subject_id` INTEGER PRIMARY KEY,
`subject_name` VARCHAR(120)
)


Table 4: Tablename is Courses
Display name: courses
DDL query for Courses:
CREATE TABLE `Courses` (
`course_id` INTEGER PRIMARY KEY,
`author_id` INTEGER NOT NULL,
`subject_id` INTEGER NOT NULL,
`course_name` VARCHAR(120),
`course_description` VARCHAR(255),
FOREIGN KEY (`author_id` ) REFERENCES `Course_Authors_and_Tutors`(`author_id` ),
FOREIGN KEY (`subject_id` ) REFERENCES `Subjects`(`subject_id` )
)


Table 5: Tablename is Student_Course_Enrolment
Display name: student course enrolment
DDL query for Student_Course_Enrolment:
CREATE TABLE `Student_Course_Enrolment` (
`registration_id` INTEGER PRIMARY KEY,
`student_id` INTEGER NOT NULL,
`course_id` INTEGER NOT NULL,
`date_of_enrolment` DATETIME NOT NULL,
`date_of_completion` DATETIME NOT NULL,
FOREIGN KEY (`course_id` ) REFERENCES `Courses`(`course_id` ),
FOREIGN KEY (`student_id` ) REFERENCES `Students`(`student_id` )
)


Table 6: Tablename is Student_Tests_Taken
Display name: student tests taken
DDL query for Student_Tests_Taken:
CREATE TABLE `Student_Tests_Taken` (
`registration_id` INTEGER NOT NULL,
`date_test_taken` DATETIME NOT NULL,
`test_result` VARCHAR(255),
FOREIGN KEY (`registration_id` ) REFERENCES `Student_Course_Enrolment`(`registration_id` )
)

This is a Database called tv_shows.
This Database consists of 5 tables.

1. city_channel (displayed as: city channel)

2. radio

3. tv_show (displayed as: tv show)

4. city_channel_radio (displayed as: city channel radio)

5. city_channel_tv_show (displayed as: city channel tv show)


Table 1: Tablename is city_channel
Display name: city channel
DDL query for city_channel:
CREATE TABLE "city_channel" (
"ID" int,
"City" text,

"Owned_Since" real,
"Affiliation" text,
PRIMARY KEY ("ID")
)

Table 4: Tablename is city_channel_radio
Display name: city channel radio
DDL query for city_channel_radio:
CREATE TABLE "city_channel_radio" (
"City_channel_ID" int,
"Radio_ID" int,

PRIMARY KEY ("City_channel_ID","Radio_ID"),

Table 5: Tablename is city_channel_tv_show

CREATE TABLE "city_channel_tv_show" (

This is a Database called tracking_share_transactions.
This Database consists of 7 tables.

These tables are:

1. Investors (displayed as: investors)

2. Lots (displayed as: lots)

3. Ref_Transaction_Types (displayed as: reference transaction types)

4. Transactions (displayed as: transactions)

5. Sales (displayed as: sales)

6. Purchases (displayed as: purchases)

7. Transactions_Lots (displayed as: transactions lots)


Table 1: Tablename is Investors
Display name: investors
DDL query for Investors:
CREATE TABLE `Investors` (
`investor_id` INTEGER PRIMARY KEY,
`Investor_details` VARCHAR(255)
)


Table 2: Tablename is Lots
Display name: lots
DDL query for Lots:
CREATE TABLE `Lots` (
`lot_id` INTEGER PRIMARY KEY,
`investor_id` INTEGER NOT NULL,
`lot_details` VARCHAR(255),
FOREIGN KEY (`investor_id` ) REFERENCES `Investors`(`investor_id` )
)


Table 3: Tablename is Ref_Transaction_Types
Display name: reference transaction types
DDL query for Ref_Transaction_Types:
CREATE TABLE `Ref_Transaction_Types` (
`transaction_type_code` VARCHAR(10) PRIMARY KEY,
`transaction_type_description` VARCHAR(80) NOT NULL
)


Table 4: Tablename is Transactions
Display name: transactions
DDL query for Transactions:
CREATE TABLE `Transactions` (
`transaction_id` INTEGER PRIMARY KEY,
`investor_id` INTEGER NOT NULL,
`transaction_type_code` VARCHAR(10) NOT NULL,
`date_of_transaction` DATETIME,
`amount_of_transaction` DECIMAL(19,4),
`share_count` VARCHAR(40),
`other_details` VARCHAR(255),
FOREIGN KEY (`investor_id` ) REFERENCES `Investors`(`investor_id` ),FOREIGN KEY (`transaction_type_code` ) REFERENCES `Ref_Transaction_Types`(`transaction_type_code` )
)


Table 5: Tablename is Sales
Display name: sales
DDL query for Sales:
CREATE TABLE `Sales` (
`sales_transaction_id` INTEGER PRIMARY KEY,
`sales_details` VARCHAR(255),
FOREIGN KEY (`sales_transaction_id` ) REFERENCES `Transactions`(`transaction_id` )
)


Table 6: Tablename is Purchases
Display name: purchases
DDL query for Purchases:
CREATE TABLE `Purchases` (
`purchase_transaction_id` INTEGER NOT NULL,
`purchase_details` VARCHAR(255) NOT NULL,
FOREIGN KEY (`purchase_transaction_id` ) REFERENCES `Transactions`(`transaction_id` )
)


Table 7: Tablename is Transactions_Lots
Display name: transactions lots
DDL query for Transactions_Lots:
CREATE TABLE `Transactions_Lots` (
`transaction_id` INTEGER NOT NULL,
`lot_id` INTEGER NOT NULL,
FOREIGN KEY (`lot_id` ) REFERENCES `Lots`(`lot_id` ),
FOREIGN KEY (`transaction_id` ) REFERENCES `Transactions`(`transaction_id` )
)

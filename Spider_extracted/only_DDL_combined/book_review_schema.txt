DDL query for book:
CREATE TABLE "book" (
"Book_ID" int,
"Title" text,
"Type" text,
"Pages" int,
"Chapters" int,

)


Table 2: Tablename is review

CREATE TABLE "review" (
"Review_ID" int,
"Book_ID" int,
"Rating" real,
"Readers_in_Million" real,
"Rank" int,
PRIMARY KEY ("Review_ID"),

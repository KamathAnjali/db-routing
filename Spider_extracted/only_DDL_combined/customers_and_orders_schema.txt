This is a Database called customers_and_orders.
This Database consists of 5 tables.

These tables are:

1. Products (displayed as: products)

2. Addresses (displayed as: addresses)

3. Customers (displayed as: customers)

4. Customer_Orders (displayed as: customer orders)

5. Order_Items (displayed as: order items)


Table 1: Tablename is Products
Display name: products
DDL query for Products:
CREATE TABLE `Products` (
`product_id` INTEGER PRIMARY KEY,
`product_type_code` VARCHAR(15),
`product_name` VARCHAR(80),
`product_price` DOUBLE NULL
)


Table 2: Tablename is Addresses
Display name: addresses
DDL query for Addresses:
CREATE TABLE `Addresses` (
`address_id` INTEGER PRIMARY KEY,
`address_details` VARCHAR(255)
)


Table 3: Tablename is Customers
Display name: customers
DDL query for Customers:
CREATE TABLE `Customers` (
`customer_id` INTEGER PRIMARY KEY,
`address_id` INTEGER NOT NULL,
`payment_method_code` VARCHAR(15),
`customer_number` VARCHAR(20),
`customer_name` VARCHAR(80),
`customer_address` VARCHAR(255),
`customer_phone` VARCHAR(80),
`customer_email` VARCHAR(80)
)


Table 4: Tablename is Customer_Orders
Display name: customer orders
DDL query for Customer_Orders:
CREATE TABLE `Customer_Orders` (
`order_id` INTEGER PRIMARY KEY,
`customer_id` INTEGER NOT NULL,
`order_date` DATETIME NOT NULL,
`order_status_code` VARCHAR(15),
FOREIGN KEY (`customer_id` ) REFERENCES `Customers`(`customer_id` )
)


Table 5: Tablename is Order_Items
Display name: order items
DDL query for Order_Items:
CREATE TABLE `Order_Items` (
`order_item_id` INTEGER NOT NULL ,
`order_id` INTEGER NOT NULL,
`product_id` INTEGER NOT NULL,
`order_quantity` VARCHAR(80),
FOREIGN KEY (`order_id` ) REFERENCES `Customer_Orders`(`order_id` ),
FOREIGN KEY (`product_id` ) REFERENCES `Products`(`product_id` )
)

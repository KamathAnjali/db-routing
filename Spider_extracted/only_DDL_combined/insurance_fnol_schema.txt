Table 1: Tablename is Customers

DDL query for Customers:
CREATE TABLE Customers (
Customer_ID INTEGER NOT NULL,
Customer_name VARCHAR(40),
PRIMARY KEY (Customer_ID)
)


Table 2: Tablename is Services
Display name: services
DDL query for Services:
CREATE TABLE Services (
Service_ID INTEGER NOT NULL,
Service_name VARCHAR(40),
PRIMARY KEY (Service_ID)
)


Table 3: Tablename is Available_Policies
Display name: available policies
DDL query for Available_Policies:
CREATE TABLE Available_Policies (
Policy_ID INTEGER NOT NULL,
policy_type_code CHAR(15),
Customer_Phone VARCHAR(255),
PRIMARY KEY (Policy_ID),
UNIQUE (Policy_ID)
)


Table 4: Tablename is Customers_Policies
Display name: customers policies
DDL query for Customers_Policies:
CREATE TABLE Customers_Policies (
Customer_ID INTEGER NOT NULL,
Policy_ID INTEGER NOT NULL,
Date_Opened DATE,
Date_Closed DATE,
PRIMARY KEY (Customer_ID, Policy_ID),
FOREIGN KEY (Customer_ID) REFERENCES Customers (Customer_ID),
FOREIGN KEY (Policy_ID) REFERENCES Available_Policies (Policy_ID)
)

CREATE TABLE First_Notification_of_Loss (

Policy_ID INTEGER NOT NULL,

PRIMARY KEY (FNOL_ID),
UNIQUE (FNOL_ID),

Table 6: Tablename is Claims

CREATE TABLE Claims (

Table 7: Tablename is Settlements
Display name: settlements
DDL query for Settlements:
CREATE TABLE Settlements (

2. press

Table 1: Tablename is author
DDL query for author:
CREATE TABLE "author" (
"Author_ID" int,
"Name" text,
"Age" int,
"Gender" text,
PRIMARY KEY ("Author_ID")
)


Table 2: Tablename is press
DDL query for press:
CREATE TABLE "press" (
"Press_ID" int,
"Name" text,
"Month_Profits_billion" real,
"Year_Profits_billion" real,
PRIMARY KEY ("Press_ID")
)


Table 3: Tablename is book
DDL query for book:
CREATE TABLE "book" (
"Book_ID" int,
"Title" text,
"Book_Series" text,
"Author_ID" int,
"Press_ID" int,
"Sale_Amount" text,
"Release_date" text,

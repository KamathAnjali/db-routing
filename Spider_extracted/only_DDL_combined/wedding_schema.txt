CREATE TABLE "people" (

"Church_ID" int,

)


Table 3: Tablename is wedding
DDL query for wedding:
CREATE TABLE "wedding" (
"Church_ID" int,
"Male_ID" int,

PRIMARY KEY ("Church_ID","Male_ID","Female_ID"),

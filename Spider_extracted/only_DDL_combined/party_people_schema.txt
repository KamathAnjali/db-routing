DDL query for region:
CREATE TABLE "region" (
"Region_ID" int,

)


Table 2: Tablename is party
DDL query for party:
CREATE TABLE "party" (
"Party_ID" int,
"Minister" text,
"Took_office" text,
"Left_office" text,
"Region_ID" int,
"Party_name" text,

FOREIGN KEY (`Region_ID`) REFERENCES `region`(`Region_ID`)
)


Table 3: Tablename is member

CREATE TABLE "member" (
"Member_ID" int,
"Member_Name" text,
"Party_ID" text,
"In_office" text,
PRIMARY KEY ("Member_ID"),
FOREIGN KEY (`Party_ID`) REFERENCES `party`(`Party_ID`)

Table 4: Tablename is party_events

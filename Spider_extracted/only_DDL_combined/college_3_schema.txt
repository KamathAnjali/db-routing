CREATE TABLE Faculty (

Display name: member of
DDL query for Member_of:
CREATE TABLE Member_of (
       FacID 	       INTEGER,

       Appt_Type       VARCHAR(15),
       FOREIGN KEY(FacID) REFERENCES Faculty(FacID),

)


Table 5: Tablename is Course

CREATE TABLE Course (
       CID   	    	VARCHAR(7) PRIMARY KEY,
       CName		VARCHAR(40),
       Credits		INTEGER,

       Days		VARCHAR(5),

Table 7: Tablename is Enrolled_in
Display name: enrolled in
DDL query for Enrolled_in:
CREATE TABLE Enrolled_in (
       StuID 		 INTEGER,
       CID		VARCHAR(7),

       FOREIGN KEY(StuID) REFERENCES Student(StuID),
       FOREIGN KEY(CID) REFERENCES Course(CID),
